**1. Create `src/lib/github/operations.ts`** — GitHub API operations for PR creation:

All functions take a `token: string` parameter and use `githubFetch` from `./client.ts`.

```typescript
import { githubFetch } from './client';
```

Implement these functions:

**`getHeadSha(owner, repo, branch, token)`** → `Promise<string>`
- `GET /repos/{owner}/{repo}/git/ref/heads/{branch}`
- Return `data.object.sha`
- Throw descriptive error on non-ok response

**`createBranch(owner, repo, branchName, sha, token)`** → `Promise<void>`
- `POST /repos/{owner}/{repo}/git/refs`
- Body: `{ ref: "refs/heads/{branchName}", sha }`
- Throw on non-ok

**`commitMultipleFiles(owner, repo, branch, files, commitMessage, headSha, token)`** → `Promise<string>`
- `files` is `Array<{ path: string; content: string }>`
- Step 1: Create blobs in parallel — `POST /repos/{owner}/{repo}/git/blobs` with `{ content: base64-encoded, encoding: 'base64' }` for each file. Use `Buffer.from(file.content).toString('base64')` for encoding.
- Step 2: Create tree — `POST /repos/{owner}/{repo}/git/trees` with `{ base_tree: headSha, tree: [{ path, mode: '100644', type: 'blob', sha: blobSha }] }`
- Step 3: Create commit — `POST /repos/{owner}/{repo}/git/commits` with `{ message: commitMessage, tree: treeSha, parents: [headSha] }`
- Step 4: Update ref — `PATCH /repos/{owner}/{repo}/git/refs/heads/${encodeURIComponent(branch)}` with `{ sha: commitSha }`
- Return commitSha
- IMPORTANT: URL-encode branch name in PATCH step (branch contains `/` e.g., `calmguard/pipeline-123`)

**`createPR(owner, repo, head, base, title, body, token)`** → `Promise<{ html_url: string; number: number }>`
- `POST /repos/{owner}/{repo}/pulls`
- Body: `{ title, body, head, base }`
- Return `{ html_url, number }` from response
- Throw on non-ok with descriptive message

```typescript
const stream = new ReadableStream({
  async start(controller) {
    const encoder = new TextEncoder();
    const emit = (data: unknown) => {
      controller.enqueue(encoder.encode(`data: ${JSON.stringify(data)}\n\n`));
    };
    try {
      emit({ type: 'step', step: 'Getting HEAD SHA...' });
      // ... perform operations, emit steps ...
      emit({ type: 'done', prUrl, prNumber, branchName, fileCount });
    } catch (error) {
      emit({ type: 'error', message: error.message });
    } finally {
      controller.close();
    }
  },
});
return new Response(stream, {
  headers: {
    'Content-Type': 'text/event-stream; charset=utf-8',
    'Cache-Control': 'no-cache, no-transform',
    'Connection': 'keep-alive',
  },
});
```

**PR generation function (for Pipeline PR button click):**

```typescript
const handleGeneratePipelinePR = async () => {
  if (!githubRepo) return;
  setPipelinePR({ status: 'generating', step: 'Starting...' });

  try {
    const res = await fetch('/api/github/create-pr', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        type: 'pipeline',
        owner: githubRepo.owner,
        repo: githubRepo.repo,
        filePath: githubRepo.filePath,
        fileSha: githubRepo.fileSha,
        defaultBranch: githubRepo.defaultBranch,
      }),
    });

    // Read SSE stream for step updates
    const reader = res.body!.getReader();
    const decoder = new TextDecoder();
    let buffer = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });
      const frames = buffer.split('\n\n');
      buffer = frames.pop() ?? '';

      for (const frame of frames) {
        const trimmed = frame.trim();
        if (!trimmed.startsWith('data: ')) continue;
        const data = JSON.parse(trimmed.slice(6));

        if (data.type === 'step') {
          setPipelinePR({ step: data.step });
        } else if (data.type === 'done') {
          setPipelinePR({
            status: 'open',
            prUrl: data.prUrl,
            prNumber: data.prNumber,
            branchName: data.branchName,
            fileCount: data.fileCount,
            step: undefined,
          });
        } else if (data.type === 'error') {
          setPipelinePR({ status: 'error', error: data.message, step: undefined });
        }
      }
    }
  } catch (error) {
    setPipelinePR({
      status: 'error',
      error: error instanceof Error ? error.message : 'Failed to generate PR',
      step: undefined,
    });
  }
};
```
